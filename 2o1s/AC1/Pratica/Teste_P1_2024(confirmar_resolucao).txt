void main(void)
{
    static char str[] = "Teste-Pratico-1";
    char *ms = str;
    char *pf = ms - 1;

    do {
        pf++;
    }while (*pf != '\0');

    while (ms < pf) {
        if (*ms < '0' || *ms > 'z') {
            *ms = '?';
        } else {
            *ms = *ms ^ 0x15;
        }
        ms++;
    }
}

#define SIZE 6

void main(void)
{
    static int in[SIZE] = {56, 11, 5, 72, 11, -15};
    static int out[2];
    int k, min1, min2;

    min1 = ~(1 << 31);
    min2 = min1;

    for (k = 0; k < SIZE; k++) {
        if (in[k] < min1) {
            min2 = min1;
            min1 = in[k];
        } else {
            if (in[k] < min2 && in[k] > min1)
                min2 = in[k];
        }
    }
    out[0] = min1;
    out[1] = min2;
}
